68 bool GameMaster::loadFile() {
69  //MainWindow is closed, mapWindow is open, close the mapwWindow
70  if(mainWindow == nullptr){
71  this->mapData = mapWindow->mapData;
72  emit mapWindow->close();
73  } else {
74  //Main window is open, close the mainWindow
75  emit mainWindow->close();
76  }
77 
78  //Playable robot must be found
79  bool userFound = false;
80 
81  QJsonArray* mapData = this->mapData;
82 
83  //If there are any elements stored in vectors, clear them
84  if(this->UserData.size() > 0){
85  UserData.clear();
86  }
87  if(this->ObstacleData.size() > 0){
88  ObstacleData.clear();
89  }
90  if(this->EnemyData.size() > 0){
91  EnemyData.clear();
92  }
93 
94  //Iterate through the mapData array
95  for (const QJsonValueRef &mapValue : *mapData) {
96 
97  //Obtain the object from the array
98  QJsonObject mapObject = mapValue.toObject();
99 
100  //Obtain time limit data
101  if(mapObject.contains("timeLimit")){
102  timeLimitData = mapObject["timeLimit"].toInt();
103  }
104 
105  QString type = mapObject["cellType"].toString();
106  //Variables for x and y position
107  int x, y;
108 
109  //Obtain x and y position if exist
110  if(mapObject.contains("mapPosition xcord") && mapObject.contains("mapPosition ycord")){
111  x = mapObject["mapPosition xcord"].toInt();
112  y = mapObject["mapPosition ycord"].toInt();
113  }
114  //Store robot data
115  if (type == "Robot") {
116  userFound = true;
117  UserData.push_back(x);
118  UserData.push_back(y);
119  QJsonObject paramObject = mapObject["parameters"].toObject();
120  UserData.push_back(paramObject["orientation"].toInt());
121  UserData.push_back(paramObject["velocity"].toInt());
122  //Store obstacle data
123  } else if (type == "Obstacle") {
124  ObstacleData.push_back(std::make_pair(x, y));
125  //Store enemy data
126  } else if (type == "Enemy") {
127  EnemyData.push_back(std::vector<int>());
128  EnemyData.back().push_back(x);
129  EnemyData.back().push_back(y);
130  QJsonObject paramObject = mapObject["parameters"].toObject();
131  EnemyData.back().push_back(paramObject["distance"].toInt());
132  EnemyData.back().push_back(paramObject["orientation"].toInt());
133  EnemyData.back().push_back(paramObject["rotationAngle"].toInt());
134  EnemyData.back().push_back(paramObject["velocity"].toInt());
135  } else {
136  continue;
137  }
138  }
139 
140  //If playable robot was not found, show error message and return false
141  if(!userFound){
142  QMessageBox::critical(this->mainWindow, "Error", "No user controlled robot found in the map");
143  return false;
144  }
145  return true;
146 }
